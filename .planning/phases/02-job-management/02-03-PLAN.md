**New endpoint (add after delete_job):**
```python
@router.post("/{job_id}/transition", response_model=JobResponse)
async def transition_job_state(
    job_id: str,
    new_state: JobState,
    current_user: User = Depends(require_admin),
    tenant_id: str = Depends(get_current_tenant),
    db: AsyncSession = Depends(get_db),
):
    """
    Transition job to new state with validation (admin only).
    
    Validates that transition is allowed before updating state.
    """
    # Get job
    result = await db.execute(select(Job).where(Job.id == job_id))
    job = result.scalar_one_or_none()
    
    if not job:
        raise HTTPException(status_code=404, detail="Job not found")
    
    # Validate transition
    if not can_transition(job.state, new_state):
        raise HTTPException(
            status_code=400,
            detail=f"Invalid transition: {job.state.value} -> {new_state.value}"
        )
    
    # Update state
    job.state = new_state
    await db.commit()
    await db.refresh(job)
    return job
```
